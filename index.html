Slide 12 : Pembatasan Tipe dengan Class Abstract


abstract class Animal {
  void makeSound();
}

class Dog implements Animal {
  @override
  void makeSound() {
    print('Woof!');
  }
}

class Cat implements Animal {
  @override
  void makeSound() {
    print('Meow!');
  }
}

class AnimalCage<T extends Animal> {
  T animal;
  AnimalCage(this.animal);

  void sound() {
    animal.makeSound();
  }
}
