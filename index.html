Slide 12 : Batasan Tipe Dengan Abstract Class


abstract class Animal {
  void makeSound();
}

class Dog implements Animal {
  @override
  void makeSound() {
    print('Woof!');
  }
}

class Cat implements Animal {
  @override
  void makeSound() {
    print('Meow!');
  }
}

class AnimalCage {
  T animal;

  AnimalCage(this.animal);

  void sound() {
    animal.makeSound();
  }
}

Slide 13 : Generic vs Dynamic vs Object
